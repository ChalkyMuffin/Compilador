program fibonacci;

var n : int;
var resultado : int;

void fibo(n:int) [
    var res1 : int;
    var res2 : int;
    var temp : int;

    {
        if (n < 2) {
            resultado = n;
        } else {
            temp = n - 1;
            fibo(temp);
            res1 = resultado;

            temp = n - 2;
            fibo(temp);
            res2 = resultado;

            resultado = res1 + res2;
        }
    }
];

main {
    n = 6;
    fibo(n);
    print(resultado);
}
end
